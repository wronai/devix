<!-- ===== devix/jetbrains-plugin/devix.properties =====
# Plugin configuration
devix.server.port=63342
devix.gateway.port=63343
devix.sync.enabled=true
devix.sync.interval=5000
devix.ai.assistant.enabled=false
devix.inspections.enabled=true
devix.quick.fixes.enabled=true
devix.notifications.enabled=true
devix.debug.mode=false

# ===== devix/jetbrains-plugin/install.sh =====
#!/bin/bash
# Instalacja wtyczki devix w JetBrains IDE

echo "🔧 Instalacja wtyczki devix dla JetBrains..."

# Znajdź katalog wtyczek
if [[ "$OSTYPE" == "darwin"* ]]; then
    PLUGIN_DIR="$HOME/Library/Application Support/JetBrains/PyCharm*/plugins"
elif [[ "$OSTYPE" == "linux-gnu"* ]]; then
    PLUGIN_DIR="$HOME/.local/share/JetBrains/PyCharm*/plugins"
elif [[ "$OSTYPE" == "msys" ]] || [[ "$OSTYPE" == "cygwin" ]]; then
    PLUGIN_DIR="$APPDATA/JetBrains/PyCharm*/plugins"
else
    echo "❌ Nieobsługiwany system operacyjny"
    exit 1
fi

# Skopiuj wtyczkę
mkdir -p "$PLUGIN_DIR/devix"
cp -r META-INF "$PLUGIN_DIR/devix/"
cp devix.properties "$PLUGIN_DIR/devix/"

echo "✅ Wtyczka zainstalowana. Restart IDE wymagany."
echo ""
echo "Aby aktywować:"
echo "1. Restart PyCharm/IntelliJ"
echo "2. Settings → Plugins → Installed → Enable devix"
echo "3. Restart IDE ponownie"
